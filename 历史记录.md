void draw_weather_graph() {
    // ========== 1. 定义绘图区域 ==========
    int graph_x = 10;       // 左边距
    int graph_y = 30;       // 顶部边距（留给状态栏）
    int graph_w = 300;      // 曲线宽度
    int graph_h = 110;      // 曲线高度
    
    // ========== 2. 清空曲线区域 ==========
    tft.fillRect(graph_x - 5, graph_y - 5, graph_w + 10, graph_h + 10, COLOR_BG);
    
    // ========== 3. 绘制坐标轴 ==========
    tft.drawRect(graph_x - 5, graph_y - 5, graph_w + 10, graph_h + 10, COLOR_AXIS);
    
    // ========== 4. 绘制网格线 ==========
    tft.setTextColor(COLOR_GRID);
    tft.setTextSize(1);
    
    // 水平网格（每20像素一条）
    for (int y = graph_y; y <= graph_y + graph_h; y += 20) {
        tft.drawLine(graph_x, y, graph_x + graph_w, y, COLOR_GRID);
    }
    // 垂直网格（每50像素一条）
    for (int x = graph_x; x <= graph_x + graph_w; x += 50) {
        tft.drawLine(x, graph_y, x, graph_y + graph_h, COLOR_GRID);
    }
    
    // ========== 5. 如果没有数据 ==========
    if (history_count < 2) {
        tft.setTextColor(TFT_WHITE);
        tft.setCursor(graph_x + 80, graph_y + 50);
        tft.println("等待数据...");
        return;
    }
    
    // ========== 6. 计算归一化范围 ==========
    float temp_min = 99, temp_max = -99;
    float humi_min = 99, humi_max = -99;
    float press_min = 9999, press_max = -9999;
    
    int count = min(history_count, MAX_POINTS);
    int start = max(0, history_count - MAX_POINTS);
    
    for (int i = start; i < start + count; i++) {
        temp_min = min(temp_min, history[i].temperature);
        temp_max = max(temp_max, history[i].temperature);
        humi_min = min(humi_min, history[i].humidity);
        humi_max = max(humi_max, history[i].humidity);
        press_min = min(press_min, history[i].pressure);
        press_max = max(press_max, history[i].pressure);
    }
    
    // 防止除零
    if (temp_max == temp_min) temp_max = temp_min + 1;
    if (humi_max == humi_min) humi_max = humi_min + 1;
    if (press_max == press_min) press_max = press_min + 1;
    
    // ========== 7. 绘制三条曲线 ==========
    for (int i = 0; i < count - 1; i++) {
        int idx1 = start + i;
        int idx2 = start + i + 1;
        
        // X坐标（均匀分布）
        int x1 = graph_x + (i * graph_w) / (count - 1);
        int x2 = graph_x + ((i + 1) * graph_w) / (count - 1);
        
        // ---- 温度曲线 (红色) ----
        int y1_temp = graph_y + graph_h - (int)((history[idx1].temperature - temp_min) * graph_h / (temp_max - temp_min));
        int y2_temp = graph_y + graph_h - (int)((history[idx2].temperature - temp_min) * graph_h / (temp_max - temp_min));
        tft.drawLine(x1, y1_temp, x2, y2_temp, COLOR_TEMP);
        // 每3个点画一个小圆点（突出数据）
        if (i % 3 == 0) {
            tft.fillCircle(x1, y1_temp, 2, COLOR_TEMP);
        }
        
        // ---- 湿度曲线 (青色) ----
        int y1_humi = graph_y + graph_h - (int)((history[idx1].humidity - humi_min) * graph_h / (humi_max - humi_min));
        int y2_humi = graph_y + graph_h - (int)((history[idx2].humidity - humi_min) * graph_h / (humi_max - humi_min));
        tft.drawLine(x1, y1_humi, x2, y2_humi, COLOR_HUMI);
        if (i % 3 == 0) {
            tft.fillCircle(x1, y1_humi, 2, COLOR_HUMI);
        }
        
        // ---- 气压曲线 (绿色) ----
        int y1_press = graph_y + graph_h - (int)((history[idx1].pressure - press_min) * graph_h / (press_max - press_min));
        int y2_press = graph_y + graph_h - (int)((history[idx2].pressure - press_min) * graph_h / (press_max - press_min));
        tft.drawLine(x1, y1_press, x2, y2_press, COLOR_PRESS);
        if (i % 3 == 0) {
            tft.fillCircle(x1, y1_press, 2, COLOR_PRESS);
        }
    }
    
    // ========== 8. 标注 Y 轴范围 ==========
    tft.setTextColor(TFT_WHITE);
    tft.setTextSize(1);
    char label[16];
    
    // 温度范围（左侧）
    sprintf(label, "%.0f℃", temp_max);
    tft.setCursor(graph_x - 45, graph_y + 5);
    tft.println(label);
    sprintf(label, "%.0f℃", temp_min);
    tft.setCursor(graph_x - 45, graph_y + graph_h - 5);
    tft.println(label);
    
    // 湿度范围（右侧）
    sprintf(label, "%.0f%%", humi_max);
    tft.setCursor(graph_x + graph_w + 5, graph_y + 5);
    tft.println(label);
    sprintf(label, "%.0f%%", humi_min);
    tft.setCursor(graph_x + graph_w + 5, graph_y + graph_h - 5);
    tft.println(label);
    
    // ========== 9. 标注最新数值（右上角浮窗） ==========
    int last = min(history_count - 1, MAX_POINTS - 1);
    tft.fillRoundRect(graph_x + graph_w - 160, graph_y + 2, 155, 22, 3, TFT_NAVY);
    tft.setTextColor(TFT_WHITE);
    tft.setTextSize(1);
    tft.setCursor(graph_x + graph_w - 155, graph_y + 7);
    tft.printf("T:%.1f℃ H:%.1f%%", 
               history[last].temperature, 
               history[last].humidity);
    tft.setCursor(graph_x + graph_w - 155, graph_y + 17);
    tft.printf("P:%.1fhPa", history[last].pressure);
    
    // ========== 10. 底部时间轴标注 ==========
    tft.setTextColor(COLOR_GRID);
    tft.setTextSize(1);
    // 在X轴标注3个时间点（起点、中间、终点）
    int points[] = {0, count/2, count-1};
    for (int p = 0; p < 3; p++) {
        int idx = start + points[p];
        if (idx < history_count) {
            struct tm *tm_info = localtime(&history[idx].timestamp);
            char time_str[8];
            strftime(time_str, sizeof(time_str), "%H:%M", tm_info);
            int x_pos = graph_x + (points[p] * graph_w) / (count - 1) - 10;
            tft.setCursor(x_pos, graph_y + graph_h + 12);
            tft.println(time_str);
        }
    }
}

顶部状态栏 + 底部信息
void draw_status_bar() {
    // 清空状态栏区域
    tft.fillRect(0, 0, 320, 25, TFT_NAVY);
    tft.setTextColor(TFT_WHITE);
    tft.setTextSize(1);
    
    // 左侧：时间
    time_t now = time(NULL);
    struct tm *tm_info = localtime(&now);
    char time_str[32];
    strftime(time_str, sizeof(time_str), "%Y-%m-%d %H:%M:%S", tm_info);
    tft.setCursor(5, 10);
    tft.println(time_str);
    
    // 右侧：实时数值（如果有数据）
    if (history_count > 0) {
        int last = min(history_count - 1, MAX_POINTS - 1);
        tft.setCursor(180, 10);
        tft.setTextColor(COLOR_TEMP);
        tft.printf("T:%.1f℃ ", history[last].temperature);
        tft.setTextColor(COLOR_HUMI);
        tft.printf("H:%.1f%% ", history[last].humidity);
        tft.setTextColor(COLOR_PRESS);
        tft.printf("P:%.1fhPa", history[last].pressure);
    }
}

// 模拟传感器读取（替换成你的 BME280 代码）
weather_record_t read_sensor() {
    weather_record_t record;
    record.temperature = 25.3 + random(-10, 10) / 10.0;  // 模拟数据
    record.humidity = 60.0 + random(-20, 20) / 10.0;
    record.pressure = 1013.0 + random(-50, 50) / 10.0;
    record.timestamp = time(NULL);
    return record;
}

// 定时器回调（每10分钟）
void timer_save_record(TimerHandle_t xTimer) {
    weather_record_t record = read_sensor();
    
    // 存入环形缓冲区
    if (history_count < MAX_POINTS) {
        history[history_count++] = record;
    } else {
        // 整体左移
        for (int i = 0; i < MAX_POINTS - 1; i++) {
            history[i] = history[i + 1];
        }
        history[MAX_POINTS - 1] = record;
    }
    
    // 写入 SPIFFS（之前实现的 append_record）
    // append_record(&record);
    
    data_updated = true;
    Serial.printf("记录 #%d: T=%.1f H=%.1f P=%.1f\n", 
                  history_count, record.temperature, 
                  record.humidity, record.pressure);
}

    tft.fillScreen(TFT_BLACK);
    
    // 2. 初始化 NVS + SPIFFS（用于存储）
    // ... 你的存储初始化代码 ...
    
    // 3. 连接 Wi-Fi + NTP 同步时间
    // ... 你的网络初始化代码 ...
    
    // 4. 先添加一些模拟数据（演示用）
    for (int i = 0; i < 30; i++) {
        weather_record_t r = read_sensor();
        history[history_count++] = r;
        delay(100);
    }
    
    // 5. 启动定时器（每10分钟）
    TimerHandle_t timer = xTimerCreate(
        "WeatherTimer", 
        SAVE_INTERVAL * 1000 / portTICK_PERIOD_MS, 
        pdTRUE, 
        NULL, 
        timer_save_record
    );
    xTimerStart(timer, 0);
    
    data_updated = true;
}

void loop() {
    if (data_updated) {
        // 一次性刷新所有界面
        draw_status_bar();
        draw_weather_graph();
        draw_bottom_bar();
        data_updated = false;
    }
    delay(100);
}
